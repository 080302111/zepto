Perform an Ajax POST request. This is a shortcut for the [$.ajax](#$.ajax)
method.

{% highlight js %}
$.post('/create', { sample: 'payload' }, function(response){
  // process response
})
{% endhighlight %}

`data` can also be a string:
{% highlight js %}
$.post('/create', $('#some_form').serialize(), function(response){
  // ...
})
{% endhighlight %}
