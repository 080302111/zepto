title: $.extend

---

Extend target object with properties from each of the source objects,
overriding the properties on target.

{% highlight js hl_lines=4 %}
var target = { one: 'patridge' },
    source = { two: 'turtle doves' }

$.extend(target, source)
//=> { one: 'patridge',
//     two: 'turtle doves' }
{% endhighlight %}
